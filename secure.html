Vulnerability Example (Remediated)


# Vulnerable code (SQL Injection)
def search_products(query):
    cursor.execute("SELECT * FROM products WHERE name LIKE '%" + query + "%'")
    return cursor.fetchall()

# Secure code (using parameterized queries)
def search_products(query):
    cursor.execute("SELECT * FROM products WHERE name LIKE %s", ('%' + query + '%',))
    return cursor.fetchall()

Vulnerability Example (Remediated)


// Vulnerable code (Excessive data exposure)
app.get('/api/users/:id', (req, res) => {
  const user = findUserById(req.params.id);
  return res.json(user); // Returns all user data including sensitive fields
});

// Secure code (Data filtering)
app.get('/api/users/:id', (req, res) => {
  const user = findUserById(req.params.id);
  
  // Only return necessary fields
  return res.json({
    id: user.id,
    name: user.name,
    email: user.email,
    // Sensitive fields like password hash, full address, etc. are excluded
  });
});

Vulnerability Example (Remediated)


// Vulnerable code (Insecure data storage)
public void saveUserCredentials(String username, String password) {
    SharedPreferences prefs = getSharedPreferences("app_prefs", MODE_PRIVATE);
    SharedPreferences.Editor editor = prefs.edit();
    editor.putString("username", username);
    editor.putString("password", password); // Storing password in plaintext
    editor.apply();
}

// Secure code (Encrypted storage)
public void saveUserCredentials(String username, String password) {
    // Using Android Keystore for encryption
    KeyStore keyStore = KeyStore.getInstance("AndroidKeyStore");
    keyStore.load(null);
    
    // Generate and use encryption key to securely store credentials
    SecretKey key = getOrCreateSecretKey();
    byte[] encryptedPassword = encrypt(password.getBytes(), key);
    
    SharedPreferences prefs = getSharedPreferences("app_prefs", MODE_PRIVATE);
    SharedPreferences.Editor editor = prefs.edit();
    editor.putString("username", username);
    editor.putString("enc_password", Base64.encodeToString(encryptedPassword, Base64.DEFAULT));
    editor.apply();
}
